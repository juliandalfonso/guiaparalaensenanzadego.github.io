  import "fmt"
  
  func main() {
  
    var edad int
    fmt.Println("Ingrese su edad")
    fmt.Scanf("%d", &edad)
    if edad >= 17 && edad <= 58 {
      fmt.Println("Estás en edad de trabajar")
    } else if edad < 17 {
      fmt.Println("Eres muy joven para trabajar")
    } else {
      fmt.Println("Estás ya jubilado")
    }
  }
